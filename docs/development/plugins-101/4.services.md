# 服务与依赖

```ts twoslash
declare module '@inpageedit/core' {
  interface InPageEdit {
    greeting(name: string): void
  }
}
// ---cut---
// 插件 A：提供了 greeting 服务
ipe.plugin((ctx) => {
  ctx.set('greeting', (name: string) => {
    console.log(`hello, ${name}`)
  })
})

// 插件 B：依赖了 greeting 服务
ipe.plugin({
  inject: ['greeting'],
  apply(ctx) {
    ctx.greeting('world') // hello, world
  },
})
```
